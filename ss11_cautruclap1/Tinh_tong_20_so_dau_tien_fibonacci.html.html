<script>
  function sumFibonacci(n) {
    let a = 0, b = 1;
    let sum = a + b; 

    for (let i = 2; i < n; i++) {
      let fib = a + b;
      sum += fib; 
      a = b; 
      b = fib;
    }

    return sum;
  }

  const total = sumFibonacci(20);
  document.getElementById('result').textContent = "Tổng của 20 số đầu tiên trong dãy Fibonacci là: " + total;
</script>

<script>
  function findFibonacciDivisibleBy5() {
    let a = 0, b = 1;
    let fib;

    while (true) {
      fib = a + b;
      if (fib % 5 === 0) {
        return fib;
      }
      a = b;
      b = fib;
    }
  }

  const result = findFibonacciDivisibleBy5();
  document.getElementById('result').textContent = "Số đầu tiên trong dãy Fibonacci chia hết cho 5 là: " + result;
</script>

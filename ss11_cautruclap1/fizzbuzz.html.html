<ul id="fizzBuzzList"></ul>

<script>
    function fizzBuzz() {
        const listElement = document.getElementById('fizzBuzzList');

        for (let i = 1; i <= 100; i++) {
            let output = '';

            if (i % 3 === 0) {
                output += 'Fizz';
            }
            if (i % 5 === 0) {
                output += 'Buzz';
            }
            if (output === '') {
                output = i; // Nếu không chia hết cho 3 hoặc 5, in ra số
            }

            let listItem = document.createElement('li');
            listItem.textContent = output;
            listElement.appendChild(listItem);
        }
    }

    fizzBuzz();
</script>

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>

<script>
  // Lấy thông tin nhiệt độ từ người dùng
  var nhietDo = parseFloat(prompt("Nhập nhiệt độ hiện tại:"));

  // Kiểm tra nhiệt độ và đưa ra thông báo
  if (nhietDo > 100) {
    alert("Nhiệt độ quá cao! Vui lòng giảm nhiệt độ.");
  } else if (nhietDo < 20) {
    alert("Nhiệt độ quá thấp! Vui lòng tăng nhiệt độ.");
  } else {
    alert("Nhiệt độ hiện tại là hợp lý.");
  }
</script>
</body>
</html>

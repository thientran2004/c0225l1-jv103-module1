<h1>20 Số Đầu Tiên Trong Dãy Fibonacci</h1>
<ul class="fibonacci-list" id="fibonacciList"></ul>

<script>
    function generateFibonacci(n) {
        let fib = [0, 1];
        for (let i = 2; i < n; i++) {
            fib[i] = fib[i - 1] + fib[i - 2];
        }
        return fib;
    }

    const fibonacciNumbers = generateFibonacci(20);
    const listElement = document.getElementById('fibonacciList');

    fibonacciNumbers.forEach(number => {
        let listItem = document.createElement('li');
        listItem.textContent = number;
        listElement.appendChild(listItem);
    });
</script>

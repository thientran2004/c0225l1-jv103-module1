<h1>Tìm Số Nguyên Nhỏ Nhất</h1>
<input type="number" id="num1" placeholder="Nhập số nguyên 1">
<input type="number" id="num2" placeholder="Nhập số nguyên 2">
<input type="number" id="num3" placeholder="Nhập số nguyên 3">
<button onclick="findMinimum()">Tìm Số Nhỏ Nhất</button>
<div class="result" id="result"></div>

<script>
    function findMinimum() {
        const num1 = parseInt(document.getElementById('num1').value);
        const num2 = parseInt(document.getElementById('num2').value);
        const num3 = parseInt(document.getElementById('num3').value);
        const minimum = Math.min(num1, num2, num3);
        document.getElementById('result').textContent = `Số nguyên nhỏ nhất là: ${minimum}`;
    }
</script>

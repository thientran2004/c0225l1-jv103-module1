<h1>Kiểm Tra Ký Tự Trong Mảng</h1>
<input type="text" id="arrayInput" placeholder="Nhập ký tự cách nhau bởi dấu phẩy">
<input type="text" id="charInput" placeholder="Nhập ký tự cần kiểm tra">
<button onclick="checkCharacter()">Kiểm Tra</button>
<div class="result" id="result"></div>

<script>
  function checkCharacter() {
    const input = document.getElementById('arrayInput').value;
    const charToCheck = document.getElementById('charInput').value;
    const charArray = input.split(',');
    let count = 0;

    charArray.forEach(char => {
      if (char.trim() === charToCheck) {
        count++;
      }
    });

    const resultDiv = document.getElementById('result');
    resultDiv.textContent = count > 0 ? `Số lần xuất hiện ký tự '${charToCheck}': ${count}` : '-1';
  }
</script>

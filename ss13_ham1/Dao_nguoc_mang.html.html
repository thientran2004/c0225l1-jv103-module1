<h1>Đảo Ngược Mảng</h1>
<input type="text" id="arrayInput" placeholder="Nhập các số nguyên, cách nhau bởi dấu phẩy">
<button onclick="reverseArray()">Đảo Ngược</button>
<div class="result" id="result"></div>

<script>
  function reverseArray() {
    const input = document.getElementById('arrayInput').value;
    const array = input.split(',').map(Number);
    const reversedArray = array.reverse();
    document.getElementById('result').textContent = `Mảng sau khi đảo ngược: ${reversedArray.join(', ')}`;
  }
</script>

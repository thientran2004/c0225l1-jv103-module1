<h1>Kiểm Tra Số Nguyên Dương</h1>
<input type="number" id="numberInput" placeholder="Nhập một số" min="1">
<button onclick="checkPositiveInteger()">Kiểm Tra</button>
<div class="result" id="result"></div>

<script>
  function checkPositiveInteger() {
    const number = parseInt(document.getElementById('numberInput').value);
    const resultDiv = document.getElementById('result');
    if (Number.isInteger(number) && number > 0) {
      resultDiv.textContent = "Số là nguyên dương.";
    } else {
      resultDiv.textContent = "Số không phải là nguyên dương.";
    }
  }
</script>

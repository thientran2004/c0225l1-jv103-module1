<h2>Tính Lãi Ngân Hàng</h2>
<input type="number" id="principalInput" placeholder="Nhập số tiền ban đầu (VNĐ)" />
<input type="number" id="monthsInput" placeholder="Nhập số tháng cho vay" />
<input type="number" id="rateInput" placeholder="Nhập lãi suất hàng tháng (%)" />
<button onclick="calculateBankInterest()">Tính Lãi</button>
<p id="result"></p>

<script>
  function calculateBankInterest() {
    const principal = parseFloat(document.getElementById('principalInput').value);
    const months = parseInt(document.getElementById('monthsInput').value);
    const rate = parseFloat(document.getElementById('rateInput').value) / 100;
    const resultElement = document.getElementById('result');

    if (isNaN(principal) || principal <= 0 || isNaN(months) || months <= 0 || isNaN(rate) || rate < 0) {
      resultElement.textContent = "Vui lòng nhập các giá trị hợp lệ.";
      return;
    }

    let totalAmount = principal * Math.pow((1 + rate), months);
    let interest = totalAmount - principal;

    resultElement.textContent = `Tổng số tiền sau ${months} tháng là: ${totalAmount.toFixed(2)} VNĐ. Lãi suất là: ${interest.toFixed(2)} VNĐ.`;
  }
</script>

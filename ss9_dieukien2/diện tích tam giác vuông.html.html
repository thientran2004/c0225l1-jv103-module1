<h2>Tính Diện Tích Tam Giác Vuông</h2>
<input type="number" id="sideAInput" placeholder="Nhập cạnh kề a" />
<input type="number" id="sideBInput" placeholder="Nhập cạnh kề b" />
<button onclick="calculateRightTriangleArea()">Tính Diện Tích</button>
<p id="result"></p>

<script>
  function calculateRightTriangleArea() {
    const a = parseFloat(document.getElementById('sideAInput').value);
    const b = parseFloat(document.getElementById('sideBInput').value);
    const resultElement = document.getElementById('result');

    if (isNaN(a) || a <= 0 || isNaN(b) || b <= 0) {
      resultElement.textContent = "Vui lòng nhập các giá trị hợp lệ cho cạnh a và b.";
      return;
    }

    const area = (a * b) / 2;
    resultElement.textContent = `Diện tích tam giác vuông có cạnh kề a = ${a} và b = ${b} là: ${area} đơn vị vuông.`;
  }
</script>

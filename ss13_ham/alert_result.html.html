<script>
    function addNumbers() {
        let firstNum = 4;
        let secondNum = 8;
        let result = firstNum + secondNum;
        return result;
    }

    function showResult() {
        let result = 0;
        alert("Giá trị của result trước khi gọi hàm: " + result); // Hiển thị giá trị trước khi gọi hàm
        result = addNumbers();
        alert("Giá trị của result sau khi gọi hàm: " + result); // Hiển thị giá trị sau khi gọi hàm
    }
</script>

<h1>Chuyển đổi Feet và Meters</h1>

<div>
  <h2>Feet sang Meters</h2>
  <input type="number" id="feetInput" placeholder="Nhập giá trị feet" />
  <button onclick="convertFeetToMeters()">Chuyển đổi</button>
  <p id="feetResult"></p>
</div>

<div>
  <h2>Meters sang Feet</h2>
  <input type="number" id="metersInput" placeholder="Nhập giá trị meters" />
  <button onclick="convertMetersToFeet()">Chuyển đổi</button>
  <p id="metersResult"></p>
</div>

<script>
  function footToMeter(foot) {
    return 0.305 * foot;
  }

  function meterToFoot(meter) {
    return 3.279 * meter;
  }

  function convertFeetToMeters() {
    const feet = parseFloat(document.getElementById('feetInput').value);
    const meters = footToMeter(feet);
    document.getElementById('feetResult').innerText = `${feet} feet bằng ${meters.toFixed(2)} meters`;
  }

  function convertMetersToFeet() {
    const meters = parseFloat(document.getElementById('metersInput').value);
    const feet = meterToFoot(meters);
    document.getElementById('metersResult').innerText = `${meters} meters bằng ${feet.toFixed(2)} feet`;
  }
</script>

<script>

  const stars = ["Polaris", "Aldebaran", "Deneb", "Vega", "Altair", "Dubhe", "Regulus"];

  const constellations = ["Ursa Minor", "Taurus", "Cygnus", "Lyra", "Aquila", "Ursa Major", "Leo"];

  function findConstellation() {

    const starName = prompt("Nhập tên ngôi sao:");


    const index = stars.indexOf(starName);
    if (index !== -1) {

      document.getElementById('result').innerText = `Chòm sao tương ứng với ${starName} là: ${constellations[index]}`;
    } else {
      document.getElementById('result').innerText = "Ngôi sao không được tìm thấy.";
    }
  }
</script>>>

<script>
  function increaseValue(number) {
    return number + 1; // Tăng giá trị lên 1
  }

  function displayIncreasedValue(value) {
    const increasedValue = increaseValue(value);
    document.getElementById('result').innerText = `Giá trị sau khi tăng: ${increasedValue}`;
  }
</script>

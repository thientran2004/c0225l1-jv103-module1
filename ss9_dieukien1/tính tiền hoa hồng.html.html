<h2>Tính Hoa Hồng Dựa Trên Doanh Số Bán Hàng</h2>
<input type="number" id="salesInput" placeholder="Nhập doanh số bán hàng" />
<button onclick="calculateCommission()">Tính Hoa Hồng</button>
<p id="result"></p>

<script>
  function calculateCommission() {
    const sales = parseFloat(document.getElementById('salesInput').value);
    const resultElement = document.getElementById('result');

    if (isNaN(sales) || sales < 0) {
      resultElement.textContent = "Vui lòng nhập một doanh số hợp lệ.";
      return;
    }

    let commissionRate;

    if (sales < 10000) {
      commissionRate = 0.05; // 5%
    } else if (sales >= 10000 && sales <= 20000) {
      commissionRate = 0.07; // 7%
    } else {
      commissionRate = 0.10; // 10%
    }

    const commission = sales * commissionRate;
    resultElement.textContent = `Hoa hồng nhận được: ${commission.toFixed(2)} VNĐ`;
  }
</script>

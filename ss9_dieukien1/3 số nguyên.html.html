<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>nhập 3 số nguyên</title>
</head>
<body>
<h2>Tìm Giá Trị Lớn Nhất Trong 3 Số</h2>
<input type="number" id="num1" placeholder="Nhập số thứ nhất" />
<input type="number" id="num2" placeholder="Nhập số thứ hai" />
<input type="number" id="num3" placeholder="Nhập số thứ ba" />
<button onclick="findMax()">Tìm Giá Trị Lớn Nhất</button>
<p id="result"></p>

<script>
  function findMax() {
    const num1 = parseInt(document.getElementById('num1').value);
    const num2 = parseInt(document.getElementById('num2').value);
    const num3 = parseInt(document.getElementById('num3').value);
    const resultElement = document.getElementById('result');

    if (isNaN(num1) || isNaN(num2) || isNaN(num3)) {
      resultElement.textContent = "Vui lòng nhập cả 3 số nguyên hợp lệ.";
      return;
    }

    const max = Math.max(num1, num2, num3);
    resultElement.textContent = `Giá trị lớn nhất là: ${max}`;
  }
</script>
</body>
</html>

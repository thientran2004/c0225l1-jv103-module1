<script>
  function calculatedays() {
      let month = +document.getElementById("month").value;
      let days;
    if(month < 1 || month > 12){
    } else {
        switch(month) {
            case 1: case 3: case 5: case 7: case 8: case 10: case 12:
                days = 31;
                break;
            case 4: case 6: case 9: case 11:
                days = 30;
                break;
            case 2:
                days = 28;
                break;
            default:
                days="ko co ngay"
        }
    }
    document.getElementById('result').innerText=`Số ngày trong tháng ${month} là ${days}`;
  }
</script>

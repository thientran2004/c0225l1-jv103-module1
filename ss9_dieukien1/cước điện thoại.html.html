    <meta charset="UTF-8">
    <title>cước điện thoại</title>
</head>
<body>
<h2>Tính Cước Điện Thoại Cho Hộ Gia Đình</h2>
<input type="number" id="minutesInput" placeholder="Nhập số phút gọi" />
<select id="packageSelect">
  <option value="basic">Gói cơ bản</option>
  <option value="standard">Gói trung bình</option>
  <option value="premium">Gói cao cấp</option>
</select>
<button onclick="calculateBill()">Tính Cước</button>
<p id="result"></p>

<script>
  function calculateBill() {
    const minutes = parseInt(document.getElementById('minutesInput').value);
    const packageType = document.getElementById('packageSelect').value;
    const resultElement = document.getElementById('result');

    if (isNaN(minutes) || minutes < 0) {
      resultElement.textContent = "Vui lòng nhập một số phút hợp lệ.";
      return;
    }

    let totalCost = 0;

    if (packageType === 'basic') {
      if (minutes <= 100) {
        totalCost = minutes * 1000;
      } else {
        totalCost = (100 * 1000) + ((minutes - 100) * 800);
      }
    } else if (packageType === 'standard') {
      if (minutes <= 200) {
        totalCost = minutes * 900;
      } else {
        totalCost = (200 * 900) + ((minutes - 200) * 700);
      }
    } else if (packageType === 'premium') {
      if (minutes <= 300) {
        totalCost = minutes * 800;
      } else {
        totalCost = (300 * 800) + ((minutes - 300) * 600);
      }
    }

    resultElement.textContent = `Cước điện thoại phải trả: ${totalCost} VNĐ`;
  }
</script>
</body>

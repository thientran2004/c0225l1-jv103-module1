<input type="number" id="testScore" placeholder="Điểm bài kiểm tra" />
<input type="number" id="midTermScore" placeholder="Điểm thi giữa kỳ" />
<input type="number" id="finalExamScore" placeholder="Điểm thi cuối kỳ" />
<button onclick="determineGrade()">Xếp Loại</button>
<p id="result"></p>

<script>
  function determineGrade() {
    const testScore = parseFloat(document.getElementById('testScore').value);
    const midTermScore = parseFloat(document.getElementById('midTermScore').value);
    const finalExamScore = parseFloat(document.getElementById('finalExamScore').value);
    const resultElement = document.getElementById('result');

    if (isNaN(testScore) || isNaN(midTermScore) || isNaN(finalExamScore)) {
      resultElement.textContent = "Vui lòng nhập tất cả các điểm hợp lệ.";
      return;
    }

    const averageScore = (testScore + midTermScore + finalExamScore) / 3;
    let grade;

    if (averageScore >= 8.5) {
      grade = "Xuất sắc";
    } else if (averageScore >= 7) {
      grade = "Giỏi";
    } else if (averageScore >= 5.5) {
      grade = "Khá";
    } else if (averageScore >= 4) {
      grade = "Trung bình";
    } else {
      grade = "Yếu";
    }

    resultElement.textContent = `Điểm trung bình: ${averageScore.toFixed(2)}, Xếp loại: ${grade}`;
  }
</script>

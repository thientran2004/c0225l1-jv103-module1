<h2>Kiểm Tra Đủ Điều Kiện Vào Lớp 10</h2>
<input type="number" id="ageInput" placeholder="Nhập tuổi của học sinh" />
<button onclick="checkEligibility()">Kiểm Tra</button>
<p id="result"></p>

<script>
  function checkEligibility() {
    const age = parseInt(document.getElementById('ageInput').value);
    const resultElement = document.getElementById('result');

    if (isNaN(age) || age < 0) {
      resultElement.textContent = "Vui lòng nhập một tuổi hợp lệ.";
      return;
    }

    if (age >= 15) {
      resultElement.textContent = "Học sinh đủ điều kiện vào học lớp 10.";
    } else {
      resultElement.textContent = "Học sinh không đủ điều kiện vào học lớp 10.";
    }
  }
</script>

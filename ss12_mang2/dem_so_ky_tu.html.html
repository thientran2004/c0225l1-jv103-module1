<h1>Đếm Ký Tự Số Trong Mảng Ký Tự</h1>
<script>
  // Khai báo mảng ký tự
  const charArray = ['a', 'b', '1', '2', 'c', '3', 'd', '4', 'e', '5'];

  // Lấy kích thước của mảng
  const size = charArray.length;

  // Hàm đếm số ký tự số
  function countDigits(array) {
    let count = 0;
    for (let char of array) {
      if (char >= '0' && char <= '9') {
        count++;
      }
    }
    return count;
  }

  // Đếm số ký tự số
  const digitCount = countDigits(charArray);

  // Hiển thị kết quả
  document.write(`<p>Kích thước của mảng: ${size}</p>`);
  document.write(`<p>Số ký tự số trong mảng: ${digitCount}</p>`);
</script>

<h1>Thay Thế Ký Tự '-' Bằng Ký Tự '_'</h1>
<input type="text" id="charArray" placeholder="Nhập mảng ký tự (cách nhau bằng dấu phẩy)" />
<br><br>
<button onclick="replaceCharacters()">Thay Thế</button>
<p id="result"></p>

<script>
  function replaceCharacters() {
    const input = document.getElementById('charArray').value;
    // Tách chuỗi thành mảng ký tự
    let charArray = input.split(',');

    // Thay thế ký tự '-' bằng '_'
    charArray = charArray.map(char => char.trim().replace(/-/g, '_'));

    // Hiển thị kết quả
    document.getElementById('result').innerText = `Mảng sau khi thay thế: ${charArray.join(', ')}`;
  }
</script>

<h1>Đếm Số Từ Trong Chuỗi</h1>
<input type="text" id="inputString" placeholder="Nhập chuỗi ở đây" />
<button onclick="countWords()">Đếm Từ</button>
<p id="result"></p>

<script>
  function countWords() {
    const input = document.getElementById('inputString').value;
    // Tách chuỗi thành mảng từ và loại bỏ các khoảng trắng thừa
    const words = input.trim().split(/\s+/);
    const wordCount = words.length;

    // Hiển thị kết quả
    document.getElementById('result').innerText = `Số từ trong chuỗi: ${wordCount}`;
  }
</script>

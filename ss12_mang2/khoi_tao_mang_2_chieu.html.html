<script>
  let a = [
    [1, 2, 1, 24],
    [8, 11, 9, 4],
    [7, 0, 7, 27],
    [7, 4, 28, 14],
    [3, 10, 26, 7]
  ];

  let output = "";

  for (let i = 0; i < a.length; i++) {
    output += "row " + i + "\n"; // In ra số hàng
    for (let j = 0; j < a[i].length; j++) {
      output += " " + a[i][j] + "\n"; // In ra phần tử
    }
  }

  document.getElementById('result').textContent = output;
</script>

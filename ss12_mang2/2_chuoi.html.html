<button onclick="compareStrings()">So Sánh</button>
<p id="result"></p>

<script>
  function compareStrings() {
    const str1 = document.getElementById('string1').value;
    const str2 = document.getElementById('string2').value;

    // So sánh hai chuỗi
    if (str1 === str2) {
      document.getElementById('result').innerText = "Hai chuỗi giống nhau.";
    } else {
      document.getElementById('result').innerText = "Hai chuỗi không giống nhau.";
    }
  }
</script>

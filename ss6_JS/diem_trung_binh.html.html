<script>
    function tinhDiemTrungBinh() {
        let ly = +document.getElementById("ly").value;
        let hoa = +document.getElementById("hoa").value;
        let sinh = +document.getElementById("sinh").value;
        let tongDiem = ly + hoa + sinh;
        let tong = tongDiem / 3;
        document.getElementById("Ket Qua").value = tong;
    }

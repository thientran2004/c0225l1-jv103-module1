<h1>Nối hai mảng số nguyên</h1>
<input type="text" id="arrayA" placeholder="Nhập mảng A (10 số, cách nhau bởi dấu phẩy)" required>
<input type="text" id="arrayB" placeholder="Nhập mảng B (10 số, cách nhau bởi dấu phẩy)" required>
<button onclick="combineArrays()">Nối mảng</button>
<p id="result"></p>

<script>
  function combineArrays() {
    const arrayAInput = document.getElementById('arrayA').value;
    const arrayBInput = document.getElementById('arrayB').value;

    const arrayA = arrayAInput.split(',').map(Number);
    const arrayB = arrayBInput.split(',').map(Number);

    // Kiểm tra độ dài của mảng A và B
    if (arrayA.length !== 10 || arrayB.length !== 10) {
      document.getElementById('result').textContent = "Vui lòng nhập đúng 10 số cho mỗi mảng.";
      return;
    }

    // Tạo mảng C bằng cách nối A và B
    const arrayC = arrayB.concat(arrayA);

    document.getElementById('result').textContent = "Mảng C: " + arrayC.join(', ');
  }
</script>

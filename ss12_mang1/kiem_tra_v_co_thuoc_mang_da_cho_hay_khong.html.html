<input type="text" id="arrayInput" placeholder="Nhập 10 số nguyên (cách nhau bởi dấu phẩy)" required>
<input type="number" id="valueInput" placeholder="Nhập số nguyên V" required>
<button onclick="removeValue()">Xóa phần tử</button>
<p id="result"></p>

<script>
  function removeValue() {
    const arrayInput = document.getElementById('arrayInput').value;
    const valueInput = parseInt(document.getElementById('valueInput').value);
    let numbers = arrayInput.split(',').map(Number);
    let message = "";

    const index = numbers.indexOf(valueInput);
    if (index !== -1) {
      // Xóa phần tử V
      for (let i = index; i < numbers.length - 1; i++) {
        numbers[i] = numbers[i + 1];
      }
      numbers[numbers.length - 1] = 0; // Gán 0 cho phần tử cuối cùng
      message = valueInput + " đã được xóa. Mảng mới: " + numbers.join(', ');
    } else {
      message = valueInput + " không có trong mảng.";
    }

    document.getElementById('result').textContent = message;
  }
</script>

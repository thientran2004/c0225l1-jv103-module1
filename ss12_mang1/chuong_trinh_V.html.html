<h1>Kiểm tra phần tử trong mảng</h1>
<input type="text" id="arrayInput" placeholder="Nhập 10 số nguyên (cách nhau bởi dấu phẩy)" required>
<input type="number" id="valueInput" placeholder="Nhập số nguyên V" required>
<button onclick="checkValue()">Kiểm tra</button>
<p id="result"></p>

<script>
  function checkValue() {
    const arrayInput = document.getElementById('arrayInput').value;
    const valueInput = parseInt(document.getElementById('valueInput').value);
    const numbers = arrayInput.split(',').map(Number);
    let message = "";

    if (numbers.includes(valueInput)) {
      message = valueInput + " is in the array.";
    } else {
      message = valueInput + " is not in the array.";
    }

    document.getElementById('result').textContent = message;
  }
</script>

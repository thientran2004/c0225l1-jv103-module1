<h1>Tìm phần tử lớn nhất trong mảng</h1>
<button onclick="findMax()">Tìm phần tử lớn nhất</button>
<p id="result"></p>
<script>
  function findMax() {
    const numbers = [5, 12, 3, 25, 8, 15, 9, 20, 7, 11];
    let max = numbers[0];
    let position = 0;

    for (let i = 1; i < numbers.length; i++) {
      if (numbers[i] > max) {
        max = numbers[i];
        position = i;
      }
    }

    document.getElementById('result').textContent = "Phần tử lớn nhất: " + max + ", Vị trí: " + position;
  }
</script>

<input type="text" id="array" placeholder="Nhập số:...">
<label for="k"></label><br>
<input type="number" id="k" min="1" placeholder="Nhập k:...">
<button onclick="countPairs">Đếm cặp</button>
<h3 id="result"></h3>
<script>
  function countPairs() {
    const arrayInput = document.getElementById('array').value;
    const k = parseInt().getElementById('k').value;
    const arr = arrayInput.split(',').map(Number);
    let count = 0;
    const n = arr.length;
    for (let i = 0; i < n; i++) {
      for (let j = i+1; j < n; j++) {
        if((arr[i] + arr[j]) % k === 0);
        count++;
      }
    }
    document.getElementById('result').innerText= `Số cặp có tổng chia hết cho k: ${count}`;
  }
</script>

<input type="text" id="numerators" placeholder="Nhập tử số (vd: 5,2,5)" />
<input type="text" id="denominators" placeholder="Nhập mẫu số (vd: 6,3,4)" />
<button onclick="findMaxFractionIndex()">Tìm Kiếm</button>
<div id="result"></div>

<script>
  function findMaxFractionIndex() {
    const numerators = document.getElementById('numerators').value.split(',').map(Number);
    const denominators = document.getElementById('denominators').value.split(',').map(Number);
    let maxIndex = 0;
    let maxValue = numerators[0] / denominators[0];

    for (let i = 1; i < numerators.length; i++) {
      const fractionValue = numerators[i] / denominators[i];
      if (fractionValue > maxValue) {
        maxValue = fractionValue;
        maxIndex = i;
      }
    }

    document.getElementById('result').textContent =
            `Chỉ số của phân số lớn nhất: ${maxIndex}`;
  }
</script>

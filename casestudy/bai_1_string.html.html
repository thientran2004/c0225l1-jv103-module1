<button onclick="processString()">Xử lý</button>

<div class="result" id="result"></div>

<script>
    function chuanHoaXau(xau) {
        let xauSach = xau.trim().replace(/\s+/g, ' ');
        return xauSach.split(' ').map(word => word.charAt(0).toUpperCase() + word.slice(1)).join(' ');
    }

    function timTuDaiNhat(xau) {
        const tuList = xau.split(' ');
        let tuDaiNhat = '';
        for (const tu of tuList) {
            if (tu.length > tuDaiNhat.length) {
                tuDaiNhat = tu;
            }
        }
        return tuDaiNhat;
    }

    function processString() {
        const inputString = document.getElementById('inputString').value;
        const xauChuanHoa = chuanHoaXau(inputString);
        const tuDaiNhat = timTuDaiNhat(inputString);
        document.getElementById('result').innerHTML =
            `Xâu chuẩn hóa: ${xauChuanHoa}<br>Từ dài nhất: ${tuDaiNhat}`;
    }
</script>

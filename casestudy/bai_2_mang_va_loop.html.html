<input type="text" id="arrayB" placeholder="Nhập mảng B (vd: 7,2,5,4,6,3)" />
<button onclick="findMissingValues()">Tìm Kiếm</button>
<div id="result"></div>

<script>
  function findMissingValues() {
    const A = document.getElementById('arrayA').value.split(',').map(Number);
    const B = document.getElementById('arrayB').value.split(',').map(Number);
    const setA = new Set(A);
    const result = [...new Set(B.filter(num => !setA.has(num)))].sort((a, b) => a - b);

    document.getElementById('result').textContent =
            `Kết quả: [${result.join(', ')}]`;
  }

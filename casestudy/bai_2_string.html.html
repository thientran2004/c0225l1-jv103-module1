<h1>Đếm số ký tự khác nhau</h1>
<textarea id="inputString" placeholder="nhập ký tự"></textarea>
<button onclick="countDifferentSymbols">Đếm</button>
<div class="result" id="result"></div>
<script>
  function countDifferentSymbols() {
    const inputString = document.getElementById('inputString').value;
    const uniqueCharacters = new Set(inputString);
    const count = uniqueCharacters.size;
    document.getElementById('result').innerHTML = `Số lượng ký tự khác nhau: ${count}`;
  }
</script>

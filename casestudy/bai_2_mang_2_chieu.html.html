<h1>Liệt kê và đếm số nguyên tố trong mảng 2 chiều</h1>
<label for="rows">Số hàng (m): </label>
<input type="number" id="rows" min="1" required>
<br>
<label for="cols">Số cột (n): </label>
<input type="number" id="cols" min="1" required>
<br>
<button onclick="createMatrix()">Nhập mảng</button>

<div id="matrixInputs" style="margin-top: 20px;"></div>

<button onclick="findPrimes()">Liệt kê số nguyên tố</button>

<div class="result" id="result"></div>

<script>
  let matrix = [];

  function createMatrix() {
    const rows = document.getElementById('rows').value;
    const cols = document.getElementById('cols').value;
    matrix = []; // Reset matrix

    let matrixInputs = document.getElementById('matrixInputs');
    matrixInputs.innerHTML = '';

    for (let i = 0; i < rows; i++) {
      let rowInputs = `<div>`;
      for (let j = 0; j < cols; j++) {
        rowInputs += `<input type="number" placeholder="a[${i}][${j}]" required>`;
      }
      rowInputs += `</div>`;
      matrixInputs.innerHTML += rowInputs;
    }
  }

  function isPrime(num) {
    if (num <= 1) return false;
    for (let i = 2; i <= Math.sqrt(num); i++) {
      if (num % i === 0) return false;
    }
    return true;
  }

  function findPrimes() {
    const inputs = document.querySelectorAll('#matrixInputs input');
    let primes = [];
    let index = 0;

    const rows = document.getElementById('rows').value;
    const cols = document.getElementById('cols').value;

    for (let i = 0; i < rows; i++) {
      for (let j = 0; j < cols; j++) {
        const value = parseInt(inputs[index].value);
        if (isPrime(value)) {
          primes.push(value);
        }
        index++;
      }
    }

    document.getElementById('result').innerHTML =
            `Số nguyên tố: ${primes.join(', ')}<br>Số lượng: ${primes.length}`;
  }
</script>

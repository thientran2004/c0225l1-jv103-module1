<h1>Tính Toán Prefix Sums</h1>
<input type="text" id="arrayInput" placeholder="Nhập mảng (vd: 1,2,3)" />
<button onclick="calculatePrefixSums()">Tính Toán</button>
<div id="result"></div>

<script>
  function calculatePrefixSums() {
    const input = document.getElementById('arrayInput').value;
    const array = input.split(',').map(Number);
    const prefixSums = [];
    let currentSum = 0;

    for (let num of array) {
      currentSum += num;
      prefixSums.push(currentSum);
    }

    document.getElementById('result').textContent =
            `Mảng prefix sums: [${prefixSums.join(', ')}]`;
  }
</script>

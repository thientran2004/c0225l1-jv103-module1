<script>
  function orderDrink() {
    const prices = [20000, 15000, 25000, 25000, 5000, 20000];
    let total = 0;
    let continueOrdering = true;

    while (continueOrdering) {
      const choice = parseInt(prompt("Chọn thức uống (1-7): "));

      if (choice === 7) {
        continueOrdering = false;
        alert("Cảm ơn bạn đã đặt hàng!");
        break;
      }

      if (choice >= 1 && choice <= 6) {
        const quantity = parseInt(prompt(`Nhập số lượng ${choice}: `));
        if (quantity > 0) {
          total += prices[choice - 1] * quantity;
          alert(`Đã thêm ${quantity} thức uống ${choice}.`);
        } else {
          alert("Số lượng phải lớn hơn 0.");
        }
      } else {
        alert("Lựa chọn không hợp lệ, vui lòng thử lại.");
      }
    }

    if (total > 0) {
      document.getElementById('result').textContent = `Tổng số tiền bạn cần phải trả: ${total} VNĐ`;
    }
  }
</script>

<input type="text" id="string2" placeholder="Nhập xâu ký tự thứ hai (s2)">
<button onclick="commonCharacterCount()">Tính</button>

<div class="result" id="result"></div>

<script>
  function commonCharacterCount() {
    const s1 = document.getElementById('string1').value;
    const s2 = document.getElementById('string2').value;

    const countMap = {};
    let commonCount = 0;

    for (const char of s1) {
      countMap[char] = (countMap[char] || 0) + 1;
    }

    for (const char of s2) {
      if (countMap[char] > 0) {
        commonCount++;
        countMap[char]--;
      }
    }

    document.getElementById('result').innerHTML =
            `Số lượng ký tự chung: ${commonCount}`;
  }
</script>

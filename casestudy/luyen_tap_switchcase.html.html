<h1>Kiểm tra nguyên âm hay phụ âm</h1>
<input type="text" id="characterInput" maxlength="1" placeholder="Nhập một ký tự" />
<button onclick="checkCharacter()">Kiểm Tra</button>
<div id="result"></div>

<script>
    function checkCharacter() {
        const input = document.getElementById('characterInput').value;
        const vowels = 'aeiouAEIOU';
        const resultDiv = document.getElementById('result');

        if (input.length !== 1) {
            resultDiv.textContent = 'Vui lòng nhập đúng một ký tự!';
            return;
        }

        if (vowels.includes(input)) {
            resultDiv.textContent = `Ký tự '${input}' là nguyên âm.`;
        } else {
            resultDiv.textContent = `Ký tự '${input}' là phụ âm.`;
        }
    }
</script>

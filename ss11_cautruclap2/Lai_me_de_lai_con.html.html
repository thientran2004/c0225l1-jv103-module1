<h1>Tính Lãi Ngân Hàng</h1>
<input type="number" id="principal" placeholder="Số tiền ban đầu" min="0" step="any">
<input type="number" id="months" placeholder="Số tháng cho vay" min="1">
<input type="number" id="interest" placeholder="Lãi suất hàng tháng (%)" min="0" step="any">
<button onclick="calculateInterest()">Tính Lãi</button>
<div id="result"></div>

<script>
  function calculateInterest() {
    const principal = parseFloat(document.getElementById('principal').value);
    const months = parseInt(document.getElementById('months').value);
    const interestRate = parseFloat(document.getElementById('interest').value) / 100;

    if (isNaN(principal) || isNaN(months) || isNaN(interestRate)) {
      document.getElementById('result').textContent = "Vui lòng nhập đầy đủ thông tin.";
      return;
    }

    let totalAmount = principal;

    for (let i = 0; i < months; i++) {
      totalAmount += totalAmount * interestRate; // Tính lãi
    }

    document.getElementById('result').textContent = `Tổng số tiền sau ${months} tháng là: ${totalAmount.toFixed(2)} VNĐ`;
  }
</script>

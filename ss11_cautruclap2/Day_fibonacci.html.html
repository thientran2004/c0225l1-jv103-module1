<script>
  function generateFibonacci() {
    const listElement = document.getElementById('fibonacci-list');
    listElement.innerHTML = ''; // Xóa danh sách hiện tại
    let n = 10; // Số lượng số trong dãy Fibonacci
    let fib = [0, 1];

    for (let i = 2; i < n; i++) {
      fib[i] = fib[i - 1] + fib[i - 2];
    }

    // Hiển thị các số trong dãy
    fib.forEach(function(num) {
      const li = document.createElement('li');
      li.textContent = num;
      listElement.appendChild(li);
    });
  }
</script>
